```powershell
$all = dir *.* -Recurse
$activity = "Replacing all instances of 'yyyy_MM_dd' with 'yyyy-MM-dd'"

$result = $all |
    foreach -Begin {
        $count = 0
    } -Process {
        Write-Progress `
            -Activity $activity `
            -PercentComplete (100 * $count / $all.Count)

        Replace-DateTimeString `
            -FilePath $_ `
            -Force

        $count++
        "Completed: $_"
    } -End {
        Write-Progress `
            -Activity $activity `
            -PercentComplete 100 `
            -Completed
    }

$result
```
